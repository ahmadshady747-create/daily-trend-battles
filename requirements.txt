# ── Core scraping library ────────────────────────────────────────────────────
# pytrends: unofficial Google Trends API wrapper
# Provides trending_searches(), interest_over_time(), etc.
pytrends==4.9.2

# ── HTTP transport (pytrends dependency, pinned for reproducibility) ─────────
requests>=2.31.0,<3.0.0

# ── Data handling (pytrends dependency) ─────────────────────────────────────
# Used internally by pytrends to parse DataFrames returned by the Trends API.
pandas>=2.0.0,<3.0.0

# ── HTML parsing (pytrends dependency) ───────────────────────────────────────
# lxml is the fastest available parser; pytrends falls back to html.parser
# but lxml avoids occasional parsing errors on malformed Trends responses.
lxml>=4.9.0,<6.0.0
